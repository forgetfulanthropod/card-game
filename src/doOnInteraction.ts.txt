addListeners()

const state = {
    interacted: false,
    callbacks: [] as Callback[]
}

function addListeners() {
    document.addEventListener("pointerdown", handleInteraction)
    // document.addEventListener("pointermove", handleInteraction)
}


function removeListeners() {
    document.removeEventListener("pointerdown", handleInteraction)
    // document.removeEventListener("pointermove", handleInteraction)
}

function handleInteraction() {
    console.log("received interaction. going through old ones.")
    state.interacted = true
    while (state.callbacks.length > 0) {
        const cb = state.callbacks.pop()
        if (cb != null) cb()
    }
    removeListeners()
}


export function addInteractionCallback(cb: Callback): void {
    if (state.interacted) {
        console.log("invoking immediately")
        cb()
    } else {
        console.log("saving for later")
        state.callbacks.push(cb)
    }
}

export function removeInteractionCallback(cb: Callback): void {
    const i = state.callbacks.indexOf(cb)
    if (i >= 0) {
        state.callbacks.splice(i, 1)
    }
}
